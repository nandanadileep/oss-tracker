# Secret / token / key patterns
# One regex per line, anchored loosely. Used by lint/secret-blocklist.py
# and scanners/repo-preflight.py. Add new patterns as you encounter them.

# GitHub personal access tokens (classic)
ghp_[A-Za-z0-9]{36,}
# GitHub fine-grained PAT
github_pat_[A-Za-z0-9_]{82,}
# GitHub OAuth / user-to-server / server-to-server / refresh
gho_[A-Za-z0-9]{36,}
ghu_[A-Za-z0-9]{36,}
ghs_[A-Za-z0-9]{36,}
ghr_[A-Za-z0-9]{36,}

# Anthropic API keys
sk-ant-[A-Za-z0-9\-]{32,}
# OpenAI keys (legacy and modern)
sk-[A-Za-z0-9]{20,}T3BlbkFJ[A-Za-z0-9]{20,}
sk-proj-[A-Za-z0-9\-_]{20,}

# AWS access keys
AKIA[0-9A-Z]{16}
ASIA[0-9A-Z]{16}

# PEM private keys
-----BEGIN (RSA |EC |OPENSSH |DSA |PGP |ENCRYPTED |)PRIVATE KEY-----

# Slack tokens
xox[baprs]-[A-Za-z0-9-]{10,}

# Stripe live keys
sk_live_[A-Za-z0-9]{24,}
rk_live_[A-Za-z0-9]{24,}

# Google API keys
AIza[0-9A-Za-z\-_]{35}

# npm tokens
npm_[A-Za-z0-9]{36}

# Vercel / Netlify / Railway / Heroku (common shape)
vercel_[A-Za-z0-9]{24,}
netlify_[A-Za-z0-9]{40,}
rqw_[A-Za-z0-9]{40,}
heroku[a-z0-9_]{8,}:[A-Fa-f0-9]{8}-[A-Fa-f0-9]{4}-[A-Fa-f0-9]{4}-[A-Fa-f0-9]{4}-[A-Fa-f0-9]{12}

# Generic password= / token= / api_key= assignments with high-entropy value
(?i)\b(password|passwd|token|api[_-]?key|secret|auth[_-]?token)\b\s*[:=]\s*['"]?[A-Za-z0-9+/=_\-]{20,}['"]?
